	# yum install -y \
    iscsi-initiator-utils.x86_64

<br/>

	# chkconfig --level 345 iscsi on
	# chkconfig --level 345 iscsid on


<br/>

Получить список экспортированных дисков с storage:

    # iscsiadm -m discovery -t sendtargets -p 192.168.3.15:3260

    Starting iscsid:                                           [  OK  ]
